public async Task ConnectAsync(ModbusConnectionOptions options, CancellationToken cancellationToken = default)
{
    await _gate.WaitAsync(cancellationToken);
    try
    {
        DisconnectCore();

        switch (options.Mode)
        {
            case ConnectionMode.Tcp:
            {
                var tcpClient = new TcpClient();
                await tcpClient.ConnectAsync(options.Host, options.Port, cancellationToken);
                tcpClient.ReceiveTimeout = options.TimeoutMilliseconds;
                tcpClient.SendTimeout = options.TimeoutMilliseconds;

                _connectionResource = tcpClient;
                _master = _factory.CreateMaster(tcpClient);
                break;
            }
            case ConnectionMode.Rtu:
            {
                var serialPort = new SerialPort(
                    options.SerialPortName,
                    options.BaudRate,
                    options.Parity,
                    options.DataBits,
                    options.StopBits)
                {
                    ReadTimeout = options.TimeoutMilliseconds,
                    WriteTimeout = options.TimeoutMilliseconds
                };

                serialPort.Open();
                var serialResource = new SerialPortStreamResource(serialPort);
                _connectionResource = serialResource;
                _master = _factory.CreateRtuMaster(serialResource);
                break;
            }
            case ConnectionMode.RtuDebugStream:
            {
                var tcpClient = new TcpClient();
                await tcpClient.ConnectAsync(options.Host, options.Port, cancellationToken);
                tcpClient.ReceiveTimeout = options.TimeoutMilliseconds;
                tcpClient.SendTimeout = options.TimeoutMilliseconds;

                var networkResource = new NetworkStreamResource(tcpClient);
                _connectionResource = networkResource;
                _master = _factory.CreateRtuMaster(networkResource);
                break;
            }
            default:
                throw new NotSupportedException($"Unsupported connection mode: {options.Mode}");
        }

        _unitId = options.UnitId;
        _master.Transport.ReadTimeout = options.TimeoutMilliseconds;
        _master.Transport.WriteTimeout = options.TimeoutMilliseconds;
        _master.Transport.Retries = 0;
    }
    catch
    {
        DisconnectCore();
        throw;
    }
    finally
    {
        _gate.Release();
    }
}
