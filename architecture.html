public GridData<TType> GetAllDataPaging(int skip, int pageSize, Expression<Func<TType, object>> keySelector,string diretion, params Expression<Func<TType, object>>[] children)
{
    List<TType> list;
    var query = dbset.AsNoTracking().AsQueryable();
    foreach (Expression<Func<TType, object>> include in children)
        query = query.Include(include);
    var count = query.Count();
    if (diretion == "down")
    {
        list = query.OrderByDescending(keySelector).Skip(skip).Take(pageSize).ToList<TType>();
    }
    else
    {
        list = query.OrderBy(keySelector).Skip(skip).Take(pageSize).ToList<TType>();
    }
    return new GridData<TType>
    {
        TotalCount = count,
        Items = list
    };
}

public async Task<List<UsageReadingByDay>> GetDailyUsageReadingByDeviceId(usageReading usageReading)
{
    List<UsageReadingByDay> list = new List<UsageReadingByDay>();
    if (usageReading.meter != null)
    {
        if (usageReading.fromDate != null && usageReading.toDate != null)
        {
            var meter = await watermeterUsageDailyRepository.GetAllWithEntities(x => x.device_id == usageReading.meter && (x.usage_date.Date >= usageReading.fromDate.Value.Date && x.usage_date.Date <= usageReading.toDate.Value.Date))
                .OrderByDescending(y => y.usage_date.Date)
                .GroupBy(z => z.usage_date.Date)
                .Select(x => new
                {
                    date = x.Key,
                    sum = x.Sum(y => y.water_usage)
                }).ToListAsync();

            list = meter.OrderBy(x => x.date)
                .Select(x => new UsageReadingByDay
                {
                    date = x.date.ToString("dd MMM yyyy"),
                    sum = x.sum
                })
                .ToList();
        }
        else
        {
            var meter = await watermeterUsageDailyRepository.GetAllWithEntities(x => x.device_id == usageReading.meter)
                .OrderByDescending(y => y.usage_date.Date)
                .GroupBy(z => z.usage_date.Date).Select(x => new
            {
                date = x.Key,
                sum = x.Sum(y => y.water_usage)
            }).ToListAsync();
            list = meter.OrderByDescending(t => t.date)
                .Take(7)
                .OrderBy(x => x.date)
                .Select(x => new UsageReadingByDay
                {
                    date = x.date.ToString("dd MMM yyyy"),
                    sum = x.sum
                })
                .ToList();
        }
        return list;
    }
    return null;

public static class PredicateBuilder
{
    public static Expression<Func<T, bool>> True<T>() { return f => true; }

    public static Expression<Func<T, bool>> False<T>() { return f => false; }

    public static Expression<Func<T, bool>> Or<T>(this Expression<Func<T, bool>> exp, Expression<Func<T, bool>> condition)
    {
        var inv = Expression.Invoke(condition, exp.Parameters.Cast<Expression>());
        return Expression.Lambda<Func<T, bool>>(Expression.Or(exp.Body, inv), exp.Parameters);
    }

    public static Expression<Func<T, bool>> And<T>(this Expression<Func<T, bool>> exp, Expression<Func<T, bool>> condition)
    {
        var inv = Expression.Invoke(condition, exp.Parameters.Cast<Expression>());
        return Expression.Lambda<Func<T, bool>>(Expression.And(exp.Body, inv), exp.Parameters);
    }
}

public async Task<Tuple<string,bool>> GetReturnDataAsync(string data, string stepFinish)
{
    switch (stepFinish)
    {
        case "HandshakeACK":
            GPRSAddress = new string(data.Skip(HEADLEN).Skip(DEVICETYPELEN).Take(ADDRESSLEN).ToArray());
            ValidateGPRSIfRegisted();
            appClient = gprsService.GetAppClientByAddress(GPRSAddress);
            gprsService.GetDownLinksByAddress(GPRSAddress).ForEach(x => downLinkQueue.Enqueue(x));
            SaveGPRSInfo(data);
            return new Tuple<string,bool>(CreateTimingPack(),true);
        case "Time-synACK":
            return new Tuple<string, bool>(CreateGPRSPack(),true);
        case "ReadingACK":
            var unPackData = UnPackExtractData(data);
            await ThirdPartyService(unPackData);
            if(downLinkQueue.Any())
            {
                downLinkQueue.TryDequeue(out GPRSDownLink downLink);
                gprsService.DeleteDownLink(downLink.DownlinkMsgId);
                return new Tuple<string, bool>(CreateGPRSPack(downLink.DownLinkMsg),false);
            }
            return new Tuple<string, bool>(CreateClosePack(),true);
        case "CloseACK":
            return new Tuple<string, bool>("Quit",true);
        default:
            return null;
    }
}

private string CreateGPRSPack(string rawData = null)
{
    var packTemplate = "7E60AAAAAAAAAAAAAA00F0000030";
    var payloadTemplate = "00000001000000000000226816200002FE0400001E0000000000810000000010";

    string cj188;
    if (rawData is null)
    {
        var cj188WaitForVerfier = $"6810{GPRSAddress.LSBReverse()}0103901F00";
        int cj188sum = 0;
        for (int i = 0; i < cj188WaitForVerfier.Length; i += 2)
        {
            cj188sum += int.Parse(cj188WaitForVerfier.Substring(i, 2), System.Globalization.NumberStyles.HexNumber);
        }
        var cj188Sum = Convert.ToString(cj188sum, 16).ToUpper();
        cj188Sum = cj188Sum.Length <= 2 ? cj188Sum.PadLeft(2, '0') : cj188Sum.Substring(cj188Sum.Length - 2, 2);
        cj188 = $"{cj188WaitForVerfier}{cj188Sum}16";
    }
    else
    {
        cj188 = rawData;
    }
    var waitForVerfier = $"{packTemplate}{payloadTemplate}{cj188}";
    int sum = 0;
    for (int i = 0; i < waitForVerfier.Length; i += 2)
    {
        sum += int.Parse(waitForVerfier.Substring(i, 2), System.Globalization.NumberStyles.HexNumber);
    }
    var atSum = Convert.ToString(sum, 16).ToUpper();
    atSum = atSum.Length <= 4 ? atSum.PadLeft(4, '0') : atSum.Substring(atSum.Length - 4, 4);

    return $"{waitForVerfier}{atSum}16";
}

public virtual string GetAtString(string hexStr)
{
    var hexArr = hexStr.Split('-').ToList();
    int countFe = 0;
    for (int i = 0; i < hexArr.Count; i++)
    {
        if (hexArr[i].ToUpper() != "FE")
        {
            countFe = i + 1;
            break;
        }
    }
    if (hexArr.Contains("68") && hexArr.Contains("16") && hexArr.Last().Equals("16") && hexArr.Count > countFe + 12)
    {
        int startDataIndex = hexArr.IndexOf("68") + 10;
        var dataLen = int.Parse(hexArr[startDataIndex], System.Globalization.NumberStyles.HexNumber);
        var lastIndexStopBit = dataLen + startDataIndex + 3;
        if (lastIndexStopBit > hexArr.Count) return null;

        if (hexArr.Count < startDataIndex + 1) return null;
        int dataLength = Convert.ToInt32(hexArr[startDataIndex], 16);
        if (hexArr.Count < startDataIndex + dataLength + 1) return null;
        var atStr = "";
        bool start = false;

        var fixBit = 0;
        if (hexArr[hexArr.Count - 3].ToUpper().Equals("0D"))
        {
            fixBit = 1;
        }

        var endIndex = startDataIndex + dataLength + fixBit;
        for (int i = 0; i < endIndex; i++)
        {
            if (hexArr[i] == "68")
            {
                start = true;
                i += 13;
                continue;
            }
            if (start)
            {
                atStr = string.IsNullOrEmpty(atStr) ? hexArr[i] : $"{atStr}-{hexArr[i]}";
            }
        }
        return IController.Hex2String(atStr);
    }
    return null;
}
