public sealed class WorkOrderService : IWorkOrderService
{
    private readonly IWorkOrderRepository _workOrderRepository;

    public WorkOrderService(IWorkOrderRepository workOrderRepository)
    {
        _workOrderRepository = workOrderRepository;
    }

    public async Task<WorkOrderDto> UpsertAsync(UpsertWorkOrderCommand command, CancellationToken cancellationToken = default)
    {
        var entity = new WorkOrder
        {
            WorkOrderNo = command.WorkOrderNo.Trim().ToUpperInvariant(),
            SkuCode = command.SkuCode.Trim().ToUpperInvariant(),
            RegionCode = command.RegionCode.Trim().ToUpperInvariant(),
            RequiredFirmwareVersion = string.IsNullOrWhiteSpace(command.RequiredFirmwareVersion) ? null : command.RequiredFirmwareVersion.Trim(),
            PlannedQuantity = command.PlannedQuantity,
            Status = command.Status,
            UpdatedAt = DateTimeOffset.UtcNow
        };

        await _workOrderRepository.UpsertAsync(entity, cancellationToken);
        return ToDto(entity);
    }

    public async Task<WorkOrderDto?> GetByNoAsync(string workOrderNo, CancellationToken cancellationToken = default)
    {
        var entity = await _workOrderRepository.GetByNoAsync(workOrderNo.Trim(), cancellationToken);
        return entity is null ? null : ToDto(entity);
    }
}
