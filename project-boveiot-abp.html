public async Task<RuleWorkflowEvaluationResultDto> EvaluateAsync(RuleWorkflowEvaluationRequestDto input)
{
    if (string.IsNullOrWhiteSpace(input.WorkflowName))
    {
        throw new ArgumentException("WorkflowName is required.", nameof(input));
    }

    if (input.Rules.Count == 0)
    {
        throw new ArgumentException("At least one rule is required.", nameof(input));
    }

    var workflow = new Workflow
    {
        WorkflowName = input.WorkflowName,
        Rules = input.Rules.Select(MapRule).ToArray()
    };

    var rulesEngine = new RulesEngine.RulesEngine(new[] { workflow });
    var parameters = input.Inputs
        .Select(item => new RuleParameter(item.Key, NormalizeInput(item.Value)!))
        .ToArray();

    var executionResults = await rulesEngine.ExecuteAllRulesAsync(input.WorkflowName, parameters);

    return new RuleWorkflowEvaluationResultDto
    {
        WorkflowName = input.WorkflowName,
        IsSuccess = executionResults.All(result => result.IsSuccess),
        EvaluatedAt = DateTimeOffset.UtcNow,
        Results = executionResults
            .Select(result => new RuleExecutionResultDto
            {
                RuleName = result.Rule.RuleName,
                IsSuccess = result.IsSuccess,
                ErrorMessage = result.Rule.ErrorMessage,
                ExceptionMessage = result.ExceptionMessage,
                SuccessEvent = result.Rule.SuccessEvent
            })
            .ToList()
    };
}
