def chat_json(
    self, *, model: str, messages: list[dict[str, Any]], **kwargs: Any
) -> dict[str, Any]:
    result = self.chat_completions(model=model, messages=messages, **kwargs)
    try:
        content = result["choices"][0]["message"]["content"]
    except (KeyError, IndexError, TypeError) as exc:
        raise BailianResponseError("Bailian response schema is missing message content.") from exc

    if not isinstance(content, str):
        raise BailianResponseError("Bailian message content is not a text JSON payload.")

    try:
        return json.loads(_strip_json_fence(content))
    except json.JSONDecodeError as exc:
        raise BailianResponseError(
            "Bailian returned non-JSON content for a JSON extraction task."
        ) from exc
